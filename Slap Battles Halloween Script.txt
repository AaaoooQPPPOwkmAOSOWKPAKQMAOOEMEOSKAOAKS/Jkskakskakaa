        for _,v in pairs(game.Players:GetChildren()) do
            if v ~= game.Players.LocalPlayer and v.Character then
                if game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and v.Character then
                    if game.Players.LocalPlayer.Character:FindFirstChild("entered") and game.Players.LocalPlayer.leaderstats.Glove.Value == "Killstreak" and v.Character:FindFirstChild("entered") then

                            end
                        end
                    end
                end
            end
        end
        task.wait()
    end
end})
